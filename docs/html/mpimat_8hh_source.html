327  if (len == oldlen)
328  {
329  this->m = nrows;

332  this->m_local = fml::bcutils::numroc(nrows, this->mb, this->g.myrow(), 0, this->g.nprow());
333  this->n_local = fml::bcutils::numroc(ncols, this->nb, this->g.mycol(), 0, this->g.npcol());
334 
335  fml::bcutils::descinit(this->desc, this->g.ictxt(), nrows, ncols, this->mb, this->nb, this->m_local);
336 
337  return;
338  }
339 
340  this->m_local = fml::bcutils::numroc(nrows, this->mb, this->g.myrow(), 0, this->g.nprow());
341  this->n_local = fml::bcutils::numroc(ncols, this->nb, this->g.mycol(), 0, this->g.npcol());

344  if (oldlen == 0)
345  realloc_ptr = malloc(len);

489 template <typename REAL>
490 void mpimat<REAL>::print(uint8_t ndigits, bool add_final_blank) const
491 {
492  for (len_t gi=0; gi<this->m; gi++)
493  {
494  for (len_t gj=0; gj<this->n; gj++)
495  {
496  const int pr = fml::bcutils::g2p(gi, this->mb, this->g.nprow());
497  const int pc = fml::bcutils::g2p(gj, this->nb, this->g.npcol());
498 
499  const int i = fml::bcutils::g2l(gi, this->mb, this->g.nprow());
500  const int j = fml::bcutils::g2l(gj, this->nb, this->g.npcol());
501 
502  REAL d;
503  if (this->g.rank0())
504  {
505  if (pr == 0 && pc == 0)
506  d = this->data[i + this->m_local*j];
507  else
508  this->g.recv(1, 1, &d, pr, pc);
509 
510  this->printval(d, ndigits);
511  }
512  else if (pr == this->g.myrow() && pc == this->g.mycol())
513  {
514  d = this->data[i + this->m_local*j];
515  this->g.send(1, 1, &d, 0, 0);
516  }
517  }
518 
519  this->g.printf(0, 0, "\n");
520  }
521 
522  if (add_final_blank)
523  this->g.printf(0, 0, "\n");
524 }

595 template <typename REAL>
596 void mpimat<REAL>::fill_linspace(const REAL start, const REAL stop)
597 {
598  if (start == stop)
599  this->fill_val(start);
600  else
601  {
602  const REAL v = (stop-start)/((REAL) this->m*this->n - 1);
603 
604  #pragma omp parallel for if((this->m_local)*(this->n_local) > fml::omp::OMP_MIN_SIZE)
605  for (len_t j=0; j<this->n_local; j++)
606  {
607  #pragma omp simd
608  for (len_t i=0; i<this->m_local; i++)
609  {
610  const int gi = fml::bcutils::l2g(i, this->mb, this->g.nprow(), this->g.myrow());
611  const int gj = fml::bcutils::l2g(j, this->nb, this->g.npcol(), this->g.mycol());
612 
613  this->data[i + this->m_local*j] = v*((REAL) gi + this->m*gj) + start;
614  }
615  }
616  }
617 }

620 inline void mpimat<int>::fill_linspace(const int start, const int stop)

622  if (start == stop)
623  this->fill_val(start);

631  #pragma omp simd

673 {

675

677  for (len_local_t j=0; j<n_local; j++)
678  {
679  for (len_local_t i=0; i<m_local; i++)
680  {
681  const int gi = fml::bcutils::l2g(i, this->mb, this->g.nprow(), this->g.myrow());
682  const int gj = fml::bcutils::l2g(j, this->nb, this->g.npcol(), this->g.mycol());
683 
684  if (gi == gj)
685  this->data[i + this->m_local*j] = v_d[gi % v.size()];
686  else
687  this->data[i + this->m_local*j] = 0;
688  }
689  }

771 template <typename REAL>
772 void mpimat<REAL>::diag(cpuvec<REAL> &v)
773 {
774  const len_t minmn = std::min(this->m, this->n);
775  v.resize(minmn);
776  v.fill_zero();
777  REAL *v_ptr = v.data_ptr();
778 
779  for (len_t gi=0; gi<minmn; gi++)
780  {
781  const len_local_t i = fml::bcutils::g2l(gi, this->mb, this->g.nprow());
782  const len_local_t j = fml::bcutils::g2l(gi, this->nb, this->g.npcol());
783 
784  const int pr = fml::bcutils::g2p(gi, this->mb, this->g.nprow());
785  const int pc = fml::bcutils::g2p(gi, this->nb, this->g.npcol());
786 
787  if (pr == this->g.myrow() && pc == this->g.mycol())
788  v_ptr[gi] = this->data[i + this->m_local*j];
789  }
790 
791 
792  this->g.allreduce(minmn, 1, v_ptr, 'A');
793 }
