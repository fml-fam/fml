492 template <typename REAL>
493 void fml::mpimat<REAL>::print(uint8_t ndigits, bool add_final_blank) const
494 {
495  for (len_t gi=0; gi<this->m; gi++)
496  {
497  for (len_t gj=0; gj<this->n; gj++)
498  {
499  const int pr = fml::bcutils::g2p(gi, this->mb, this->g.nprow());
500  const int pc = fml::bcutils::g2p(gj, this->nb, this->g.npcol());
501 
502  const int i = fml::bcutils::g2l(gi, this->mb, this->g.nprow());
503  const int j = fml::bcutils::g2l(gj, this->nb, this->g.npcol());
504 
505  REAL d;
506  if (this->g.rank0())
507  {
508  if (pr == 0 && pc == 0)
509  d = this->data[i + this->m_local*j];
510  else
511  this->g.recv(1, 1, &d, pr, pc);
512 
513  this->printval(d, ndigits);
514  }
515  else if (pr == this->g.myrow() && pc == this->g.mycol())
516  {
517  d = this->data[i + this->m_local*j];
518  this->g.send(1, 1, &d, 0, 0);
519  }
520  }
521 
522  this->g.printf(0, 0, "\n");
523  }
524 
525  if (add_final_blank)
526  this->g.printf(0, 0, "\n");
527 }

598 template <typename REAL>
599 void fml::mpimat<REAL>::fill_linspace(const REAL start, const REAL stop)
600 {
601  if (start == stop)
602  this->fill_val(start);
603  else
604  {
605  const REAL v = (stop-start)/((REAL) this->m*this->n - 1);
606 
607  #pragma omp parallel for if((this->m_local)*(this->n_local) > fml::omp::OMP_MIN_SIZE)
608  for (len_t j=0; j<this->n_local; j++)
609  {
610  #pragma omp simd
611  for (len_t i=0; i<this->m_local; i++)
612  {
613  const int gi = fml::bcutils::l2g(i, this->mb, this->g.nprow(), this->g.myrow());
614  const int gj = fml::bcutils::l2g(j, this->nb, this->g.npcol(), this->g.mycol());
615 
616  this->data[i + this->m_local*j] = v*((REAL) gi + this->m*gj) + start;
617  }
618  }
619  }
620 }

623 inline void fml::mpimat<int>::fill_linspace(const int start, const int stop)
624 {
625  if (start == stop)
626  this->fill_val(start);
627  else
628  {

630

632  for (len_t j=0; j<this->n_local; j++)
633  {
634  #pragma omp simd

638  const int gj = fml::bcutils::l2g(j, this->nb, this->g.npcol(), this->g.mycol());
639

641  }
642  }

774 template <typename REAL>
775 void fml::mpimat<REAL>::diag(fml::cpuvec<REAL> &v)
776 {
777  const len_t minmn = std::min(this->m, this->n);
778  v.resize(minmn);
779  v.fill_zero();
780  REAL *v_ptr = v.data_ptr();
781 
782  #pragma omp parallel for if(minmn > fml::omp::OMP_MIN_SIZE)
783  for (len_t gi=0; gi<minmn; gi++)
784  {
785  const len_local_t i = fml::bcutils::g2l(gi, this->mb, this->g.nprow());
786  const len_local_t j = fml::bcutils::g2l(gi, this->nb, this->g.npcol());
787 
788  const int pr = fml::bcutils::g2p(gi, this->mb, this->g.nprow());
789  const int pc = fml::bcutils::g2p(gi, this->nb, this->g.npcol());
790 
791  if (pr == this->g.myrow() && pc == this->g.mycol())
792  v_ptr[gi] = this->data[i + this->m_local*j];
793  }
794 
795 
796  this->g.allreduce(minmn, 1, v_ptr, 'A');
797 }

819 {
820  const len_t minmn = std::min(this->m, this->n);
821  v.resize(minmn);
822  v.fill_zero();
823  REAL *v_ptr = v.data_ptr();
824 
825  #pragma omp parallel for if(minmn > fml::omp::OMP_MIN_SIZE)
826  for (len_t gi=0; gi<minmn; gi++)
