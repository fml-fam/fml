317 template <typename REAL>
318 void mpimat<REAL>::resize(len_t nrows, len_t ncols)
319 {
320  check_params(nrows, ncols, this->mb, this->nb);
321 
322  const size_t len = (size_t) nrows * ncols * sizeof(REAL);
323  const size_t oldlen = (size_t) this->m * this->n * sizeof(REAL);
324 
325  if (len == oldlen)
326  {
327  this->m = nrows;
328  this->n = ncols;
329  return;
330  }
331 
332  this->m_local = fml::bcutils::numroc(nrows, this->mb, this->g.myrow(), 0, this->g.nprow());
333  this->n_local = fml::bcutils::numroc(ncols, this->nb, this->g.mycol(), 0, this->g.npcol());
334 
335  void *realloc_ptr;
336  if (oldlen == 0)
337  realloc_ptr = malloc(len);
338  else
339  realloc_ptr = realloc(this->data, len);
340 
341  if (realloc_ptr == NULL)
342  throw std::bad_alloc();
343 
344  this->data = (REAL*) realloc_ptr;
345 
346  fml::bcutils::descinit(this->desc, this->g.ictxt(), nrows, ncols, this->mb, this->nb, this->m_local);
347 
348  this->m = nrows;
349  this->n = ncols;
350 }

475 template <typename REAL>
476 void mpimat<REAL>::print(uint8_t ndigits, bool add_final_blank) const
477 {
478  for (len_t gi=0; gi<this->m; gi++)
479  {
480  for (len_t gj=0; gj<this->n; gj++)
481  {
482  const int pr = fml::bcutils::g2p(gi, this->mb, this->g.nprow());
483  const int pc = fml::bcutils::g2p(gj, this->nb, this->g.npcol());
484 
485  const int i = fml::bcutils::g2l(gi, this->mb, this->g.nprow());
486  const int j = fml::bcutils::g2l(gj, this->nb, this->g.npcol());
487 
488  REAL d;
489  if (this->g.rank0())
490  {
491  if (pr == 0 && pc == 0)
492  d = this->data[i + this->m_local*j];
493  else
494  this->g.recv(1, 1, &d, pr, pc);
495 
496  this->printval(d, ndigits);
497  }
498  else if (pr == this->g.myrow() && pc == this->g.mycol())
499  {
500  d = this->data[i + this->m_local*j];
501  this->g.send(1, 1, &d, 0, 0);
502  }
503  }
504 
505  this->g.printf(0, 0, "\n");
506  }
507 
508  if (add_final_blank)
509  this->g.printf(0, 0, "\n");
510 }

581 template <typename REAL>
582 void mpimat<REAL>::fill_linspace(const REAL start, const REAL stop)
583 {
584  if (start == stop)
585  this->fill_val(start);
586  else
587  {
588  const REAL v = (stop-start)/((REAL) this->m*this->n - 1);
589 
590  #pragma omp parallel for if((this->m_local)*(this->n_local) > fml::omp::OMP_MIN_SIZE)
591  for (len_t j=0; j<this->n_local; j++)
592  {
593  #pragma omp simd
594  for (len_t i=0; i<this->m_local; i++)
595  {
596  const int gi = fml::bcutils::l2g(i, this->mb, this->g.nprow(), this->g.myrow());
597  const int gj = fml::bcutils::l2g(j, this->nb, this->g.npcol(), this->g.mycol());
598 
599  this->data[i + this->m_local*j] = v*((REAL) gi + this->m*gj) + start;
600  }
601  }
602  }
603 }

759 template <typename REAL>
760 void mpimat<REAL>::diag(cpuvec<REAL> &v)
761 {
762  const len_t minmn = std::min(this->m, this->n);
763  v.resize(minmn);
764  v.fill_zero();
765  REAL *v_ptr = v.data_ptr();
766 
767  for (len_t gi=0; gi<minmn; gi++)
768  {
769  const len_local_t i = fml::bcutils::g2l(gi, this->mb, this->g.nprow());
770  const len_local_t j = fml::bcutils::g2l(gi, this->nb, this->g.npcol());
771 
772  const int pr = fml::bcutils::g2p(gi, this->mb, this->g.nprow());
773  const int pc = fml::bcutils::g2p(gi, this->nb, this->g.npcol());
774 
775  if (pr == this->g.myrow() && pc == this->g.mycol())
776  v_ptr[gi] = this->data[i + this->m_local*j];
777  }
778 
779 
780  this->g.allreduce(minmn, 1, v_ptr, 'A');
781 }

855 {
856  cpuvec<REAL> tmp(this->nb);

858

860  const int mycol = this->g.mycol();
