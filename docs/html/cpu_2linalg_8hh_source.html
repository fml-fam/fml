40  void add(const bool transx, const bool transy, const REAL alpha, const REAL beta, const cpumat<REAL> &x, const cpumat<REAL> &y, cpumat<REAL> &ret)
41  {
42  len_t m, n;
43  fml::linalgutils::matadd_params(transx, transy, x.nrows(), x.ncols(), y.nrows(), y.ncols(), &m, &n);
44 
45  if (ret.nrows() != m || ret.ncols() != n)
46  ret.resize(m, n);
47 
48  const REAL *x_d = x.data_ptr();
49  const REAL *y_d = y.data_ptr();
50  REAL *ret_d = ret.data_ptr();
51 
52  if (!transx && !transy)
53  {
54  #pragma omp parallel for if(m*n > fml::omp::OMP_MIN_SIZE)
55  for (len_t j=0; j<n; j++)
56  {
57  #pragma omp simd
58  for (len_t i=0; i<m; i++)
59  ret_d[i + m*j] = alpha*x_d[i + m*j] + beta*y_d[i + m*j];
60  }
61  }
62  else if (transx && transy)
63  {
64  #pragma omp parallel for if(m*n > fml::omp::OMP_MIN_SIZE)
65  for (len_t j=0; j<m; j++)
66  {
67  #pragma omp simd
68  for (len_t i=0; i<n; i++)
69  ret_d[j + m*i] = alpha*x_d[i + n*j] + beta*y_d[i + n*j];
70  }
71  }
72  else if (transx && !transy)
73  {
74  #pragma omp parallel for if(m*n > fml::omp::OMP_MIN_SIZE)
75  for (len_t j=0; j<n; j++)
76  {
77  #pragma omp simd
78  for (len_t i=0; i<m; i++)
79  ret_d[i + m*j] = alpha*x_d[j + n*i] + beta*y_d[i + m*j];
80  }
81  }
82  else if (!transx && transy)
83  {
84  #pragma omp parallel for if(m*n > fml::omp::OMP_MIN_SIZE)
85  for (len_t j=0; j<n; j++)
86  {

281  void xpose(const cpumat<REAL> &x, cpumat<REAL> &tx)
282  {
283  const len_t m = x.nrows();
284  const len_t n = x.ncols();
285 
286  if (m != tx.ncols() || n != tx.nrows())
287  tx.resize(n, m);
288 
289  const int blocksize = 8;
290  const REAL *x_d = x.data_ptr();
291  REAL *tx_d = tx.data_ptr();
292 
293  #pragma omp parallel for shared(tx) schedule(dynamic, 1) if(m*n>fml::omp::OMP_MIN_SIZE)
294  for (int j=0; j<n; j+=blocksize)
295  {
296  for (int i=0; i<m; i+=blocksize)
297  {
298  for (int col=j; col<j+blocksize && col<n; ++col)
299  {
300  for (int row=i; row<i+blocksize && row<m; ++row)
301  tx_d[col + n*row] = x_d[row + m*col];
302  }
303  }
304  }
305  }
