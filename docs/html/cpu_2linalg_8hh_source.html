41  template <typename REAL>
42  void add(const bool transx, const bool transy, const REAL alpha, const REAL beta, const cpumat<REAL> &x, const cpumat<REAL> &y, cpumat<REAL> &ret)
43  {
44  len_t m, n;
45  fml::linalgutils::matadd_params(transx, transy, x.nrows(), x.ncols(), y.nrows(), y.ncols(), &m, &n);
46 
47  if (ret.nrows() != m || ret.ncols() != n)
48  ret.resize(m, n);
49 
50  const REAL *x_d = x.data_ptr();
51  const REAL *y_d = y.data_ptr();
52  REAL *ret_d = ret.data_ptr();
53 
54  if (!transx && !transy)
55  {
56  #pragma omp parallel for if(m*n > fml::omp::OMP_MIN_SIZE)
57  for (len_t j=0; j<n; j++)
58  {
59  #pragma omp simd
60  for (len_t i=0; i<m; i++)
61  ret_d[i + m*j] = alpha*x_d[i + m*j] + beta*y_d[i + m*j];
62  }
63  }
64  else if (transx && transy)
65  {
66  #pragma omp parallel for if(m*n > fml::omp::OMP_MIN_SIZE)
67  for (len_t j=0; j<m; j++)
68  {
69  #pragma omp simd
70  for (len_t i=0; i<n; i++)
71  ret_d[j + m*i] = alpha*x_d[i + n*j] + beta*y_d[i + n*j];
72  }
73  }
74  else if (transx && !transy)
75  {
76  #pragma omp parallel for if(m*n > fml::omp::OMP_MIN_SIZE)
77  for (len_t j=0; j<n; j++)
78  {
79  #pragma omp simd
80  for (len_t i=0; i<m; i++)
81  ret_d[i + m*j] = alpha*x_d[j + n*i] + beta*y_d[i + m*j];
82  }
83  }
84  else if (!transx && transy)
85  {
86  #pragma omp parallel for if(m*n > fml::omp::OMP_MIN_SIZE)
87  for (len_t j=0; j<n; j++)
88  {
89  #pragma omp simd
90  for (len_t i=0; i<m; i++)
91  ret_d[i + m*j] = alpha*x_d[i + m*j] + beta*y_d[j + n*i];
92  }
93  }
94  }

284  template <typename REAL>
285  void xpose(const cpumat<REAL> &x, cpumat<REAL> &tx)
286  {
287  const len_t m = x.nrows();
288  const len_t n = x.ncols();
289 
290  if (m != tx.ncols() || n != tx.nrows())
291  tx.resize(n, m);
292 
293  const int blocksize = 8;
294  const REAL *x_d = x.data_ptr();
295  REAL *tx_d = tx.data_ptr();
296 
297  #pragma omp parallel for shared(tx) schedule(dynamic, 1) if(m*n>fml::omp::OMP_MIN_SIZE)
298  for (int j=0; j<n; j+=blocksize)
299  {
300  for (int i=0; i<m; i+=blocksize)
301  {
302  for (int col=j; col<j+blocksize && col<n; ++col)
303  {
304  for (int row=i; row<i+blocksize && row<m; ++row)
305  tx_d[col + n*row] = x_d[row + m*col];
306  }
307  }
308  }
309  }

387  template <typename REAL>
388  void det(cpumat<REAL> &x, int &sign, REAL &modulus)
389  {
390  const len_t m = x.nrows();
391  if (!x.is_square())
392  throw std::runtime_error("'x' must be a square matrix");
393 
394  cpuvec<int> p;
395  int info;
396  lu(x, p, info);
397 
398  if (info != 0)
399  {
400  if (info > 0)
401  {
402  sign = 1;
403  modulus = -INFINITY;
404  return;
405  }
406  else
407  return;
408  }
409 
410 
411  // get determinant
412  REAL mod = 0.0;
413  int sgn = 1;
414 
415  const int *ipiv = p.data_ptr();
416  for (int i=0; i<m; i++)
417  {
418  if (ipiv[i] != (i + 1))
419  sgn = -sgn;
420  }
421 
422  const REAL *a = x.data_ptr();
423 
424  #pragma omp parallel for reduction(+:mod) reduction(*:sgn)
425  for (int i=0; i<m; i++)
426  {
427  const REAL d = a[i + m*i];
428  if (d < 0)
429  {
430  mod += log(-d);
431  sgn *= -1;
432  }
433  else
434  mod += log(d);
435  }
436 
437  modulus = mod;
438  sign = sgn;
439  }

470  int svd_internals(const int nu, const int nv, cpumat<REAL> &x, cpuvec<REAL> &s,
471  cpumat<REAL> &u, cpumat<REAL> &vt)
472  {
473  int info = 0;
474  char jobz;
475  int ldvt;
476 
477  const len_t m = x.nrows();
478  const len_t n = x.ncols();
479  const len_t minmn = std::min(m, n);
480 
481  s.resize(minmn);
482 
483  if (nu == 0 && nv == 0)
484  {
485  jobz = 'N';
486  ldvt = 1; // value is irrelevant, but must exist!
487  }
488  else if (nu <= minmn && nv <= minmn)
489  {
490  jobz = 'S';
491  ldvt = minmn;
492 
493  u.resize(m, minmn);
494  vt.resize(minmn, n);
495  }
496  else
497  {
498  jobz = 'A';
499  ldvt = n;
500  }
501 
502  cpuvec<int> iwork(8*minmn);
503 
504  REAL tmp;
505  fml::lapack::gesdd(jobz, m, n, x.data_ptr(), m, s.data_ptr(), u.data_ptr(), m, vt.data_ptr(), ldvt, &tmp, -1, iwork.data_ptr(), &info);
506  int lwork = (int) tmp;
507  cpuvec<REAL> work(lwork);
508 
509  fml::lapack::gesdd(jobz, m, n, x.data_ptr(), m, s.data_ptr(), u.data_ptr(), m, vt.data_ptr(), ldvt, work.data_ptr(), lwork, iwork.data_ptr(), &info);
510 
511  return info;
512  }

582  support.data_ptr(), &worksize, -1, &liwork, -1,

585  lwork = (int) worksize;
586  cpuvec<REAL> work(lwork);
587  cpuvec<int> iwork(liwork);
588 
589  fml::lapack::syevr(jobz, 'A', 'L', n, x.data_ptr(), n, (REAL) 0.f, (REAL) 0.f,
590  0, 0, (REAL) 0.f, &nfound, values.data_ptr(), vectors.data_ptr(), n,
591  support.data_ptr(), work.data_ptr(), lwork, iwork.data_ptr(), liwork,
592  &info);
593 
594  return info;
