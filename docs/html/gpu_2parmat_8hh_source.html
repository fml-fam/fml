68 template <typename REAL>
69 void fml::parmat_gpu<REAL>::print(uint8_t ndigits, bool add_final_blank)
70 {
71  len_t n = this->data.ncols();
72  fml::gpuvec<REAL> pv(this->data.get_card(), n);
73 
74  int myrank = this->r.rank();
75  if (myrank == 0)
76  this->data.print(ndigits, false);
77 
78  for (int rank=1; rank<this->r.size(); rank++)
79  {
80  if (rank == myrank)
81  {
82  len_t m = this->data.nrows();
83  this->r.send(1, &m, 0);
84 
85  for (int i=0; i<m; i++)
86  {
87  this->data.get_row(i, pv);
88  this->r.send(n, pv.data_ptr(), 0);
89  }
90  }
91  else if (myrank == 0)
92  {
93  len_t m;
94  this->r.recv(1, &m, rank);
95 
96  for (int i=0; i<m; i++)
97  {
98  this->r.recv(n, pv.data_ptr(), rank);
99  pv.print(ndigits, false);
100  }
101  }
102 
103  this->r.barrier();
104  }
105 
106  if (add_final_blank)
107  {
108  this->r.printf(0, "\n");
109  this->r.barrier();
110  }
111 }
