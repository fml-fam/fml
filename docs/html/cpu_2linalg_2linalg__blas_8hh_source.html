44  fml::linalgutils::matadd_params(transx, transy, x.nrows(), x.ncols(), y.nrows(), y.ncols(), &m, &n);
45 
46  if (ret.nrows() != m || ret.ncols() != n)
47  ret.resize(m, n);
48 
49  const REAL *x_d = x.data_ptr();
50  const REAL *y_d = y.data_ptr();
51  REAL *ret_d = ret.data_ptr();
52 
53  if (!transx && !transy)
54  {
55  #pragma omp parallel for if(m*n > fml::omp::OMP_MIN_SIZE)
56  for (len_t j=0; j<n; j++)
57  {
58  #pragma omp simd
59  for (len_t i=0; i<m; i++)
60  ret_d[i + m*j] = alpha*x_d[i + m*j] + beta*y_d[i + m*j];
61  }
62  }
63  else if (transx && transy)
64  {
65  #pragma omp parallel for if(m*n > fml::omp::OMP_MIN_SIZE)
66  for (len_t j=0; j<m; j++)
67  {
68  #pragma omp simd
69  for (len_t i=0; i<n; i++)
70  ret_d[j + m*i] = alpha*x_d[i + n*j] + beta*y_d[i + n*j];
71  }
72  }
73  else if (transx && !transy)
74  {

77  {

283  template <typename REAL>
284  void xpose(const cpumat<REAL> &x, cpumat<REAL> &tx)
285  {
286  const len_t m = x.nrows();
287  const len_t n = x.ncols();
288 
289  if (m != tx.ncols() || n != tx.nrows())
290  tx.resize(n, m);
291 
292  const int blocksize = 8;
293  const REAL *x_d = x.data_ptr();
294  REAL *tx_d = tx.data_ptr();
295 
296  #pragma omp parallel for shared(tx) schedule(dynamic, 1) if(m*n > fml::omp::OMP_MIN_SIZE)
297  for (int j=0; j<n; j+=blocksize)
298  {
299  for (int i=0; i<m; i+=blocksize)
300  {
301  for (int col=j; col<j+blocksize && col<n; ++col)
302  {
303  for (int row=i; row<i+blocksize && row<m; ++row)
304  tx_d[col + n*row] = x_d[row + m*col];
305  }
306  }
307  }
308  }
