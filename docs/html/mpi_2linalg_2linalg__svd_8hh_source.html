31  void tssvd(mpimat<REAL> &x, cpuvec<REAL> &s, mpimat<REAL> &u, mpimat<REAL> &vt)
32  {
33  const len_t m = x.nrows();
34  const len_t n = x.ncols();
35  if (m <= n)
36  throw std::runtime_error("'x' must have more rows than cols");
37 
38  const grid g = x.get_grid();
39 
40  cpuvec<REAL> qraux(n);
41  cpuvec<REAL> work(m);
42  qr_internals(false, x, qraux, work);
43 
44  mpimat<REAL> R(g, n, n, x.bf_rows(), x.bf_cols());
45  qr_R(x, R);
46 
47  mpimat<REAL> u_R(g, n, n, x.bf_rows(), x.bf_cols());
48  svd(R, s, u_R, vt);
49 
50  u.resize(m, n);
51  u.fill_eye();
52 
53  qr_Q(x, qraux, u, work);
54 
55  matmult(false, false, (REAL)1.0, u, u_R, x);
56  copy::mpi2mpi(x, u);
57  }

239  template <typename REAL>
240  void cpsvd(const mpimat<REAL> &x, cpuvec<REAL> &s, mpimat<REAL> &u, mpimat<REAL> &vt)
241  {
242  err::check_grid(x, u);
243  err::check_grid(x, vt);
244 
245  const len_t m = x.nrows();
246  const len_t n = x.ncols();
247  const len_t minmn = std::min(m, n);
248 
249  const grid g = x.get_grid();
250  mpimat<REAL> cp(g, x.bf_rows(), x.bf_cols());
251 
252  if (m >= n)
253  {
254  crossprod((REAL)1.0, x, cp);
255  eigen_sym(cp, s, vt);
256  vt.rev_cols();
257  copy::mpi2mpi(vt, cp);
258  }
259  else
260  {
261  tcrossprod((REAL)1.0, x, cp);
262  eigen_sym(cp, s, u);
263  u.rev_cols();
264  copy::mpi2mpi(u, cp);
265  }
266 
267  s.rev();
268  REAL *s_d = s.data_ptr();
269  #pragma omp for simd
270  for (len_t i=0; i<s.size(); i++)
271  s_d[i] = sqrt(fabs(s_d[i]));
272 
273  len_t m_local, n_local;
274  REAL *ev_d;
275  if (m >= n)
276  {
277  m_local = vt.nrows_local();
278  n_local = vt.ncols_local();
279  ev_d = vt.data_ptr();
280  }
281  else
282  {
283  m_local = cp.nrows_local();
284  n_local = cp.ncols_local();
285  ev_d = cp.data_ptr();
286  }
287 
288  for (len_t j=0; j<n_local; j++)
289  {
290  #pragma omp for simd
291  for (len_t i=0; i<m_local; i++)
292  {
293  const int gi = fml::bcutils::l2g(i, x.bf_rows(), g.nprow(), g.myrow());
294  const int gj = fml::bcutils::l2g(j, x.bf_cols(), g.npcol(), g.mycol());
295 
296  if (gi < minmn && gj < minmn)
297  ev_d[i + m_local*j] /= s_d[gj];
298  }
299  }
300 
301  if (m >= n)
302  {
303  matmult(false, false, (REAL)1.0, x, vt, u);
304  xpose(cp, vt);
305  }
306  else
307  matmult(true, false, (REAL)1.0, cp, x, vt);
308  }
