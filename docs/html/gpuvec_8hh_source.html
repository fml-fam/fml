223 template <typename T>
224 void fml::gpuvec<T>::resize(len_t size)
225 {
226  check_params(size);
227 
228  if (this->_size == size)
229  return;
230 
231  size_t len = (size_t) size * sizeof(T);
232 
233  T *realloc_ptr;
234  realloc_ptr = (T*) this->c->mem_alloc(len);
235 
236  size_t oldlen = (size_t) this->_size * sizeof(T);
237  size_t copylen = std::min(len, oldlen);
238  this->c->mem_gpu2gpu(realloc_ptr, this->data, copylen);
239  this->c->mem_free(this->data);
240  this->data = realloc_ptr;
241 
242  this->_size = size;
243 
244  dim_block = fml::kernel_launcher::dim_block1();
245  dim_grid = fml::kernel_launcher::dim_grid(this->_size);
246 }

554 template <typename T>
555 bool fml::gpuvec<T>::operator==(const fml::gpuvec<T> &x) const
556 {
557  if (this->_size != x.size())
558  return false;
559  else if (this->c->get_id() != x.get_card()->get_id())
560  return false;
561  else if (this->data == x.data_ptr())
562  return true;
563 
564  int all_eq = 1;
565  fml::gpuscalar<int> all_eq_gpu(c, all_eq);
566 
567  fml::kernelfuns::kernel_all_eq<<<dim_grid, dim_block>>>(this->_size, 1, this->data, x.data_ptr(), all_eq_gpu.data_ptr());
568 
569  all_eq_gpu.get_val(&all_eq);
570  this->c->check();
571 
572  return (bool) all_eq;
573 }
