215 template <typename T>
216 void gpuvec<T>::resize(len_t size)
217 {
218  check_params(size);
219 
220  if (this->_size == size)
221  return;
222 
223  size_t len = (size_t) size * sizeof(T);
224 
225  T *realloc_ptr;
226  realloc_ptr = (T*) this->c->mem_alloc(len);
227 
228  size_t oldlen = (size_t) this->_size * sizeof(T);
229  size_t copylen = std::min(len, oldlen);
230  this->c->mem_gpu2gpu(realloc_ptr, this->data, copylen);
231  this->c->mem_free(this->data);
232  this->data = realloc_ptr;
233 
234  this->_size = size;
235 
236  dim_block = fml::kernel_launcher::dim_block1();
237  dim_grid = fml::kernel_launcher::dim_grid(this->_size);
238 }

483 template <typename T>
484 bool gpuvec<T>::operator==(const gpuvec<T> &x) const
485 {
486  if (this->_size != x.size())
487  return false;
488  else if (this->c->get_id() != x.get_card()->get_id())
489  return false;
490  else if (this->data == x.data_ptr())
491  return true;
492 
493  int all_eq = 1;
494  gpuscalar<int> all_eq_gpu(c, all_eq);
495 
496  fml::kernelfuns::kernel_all_eq<<<dim_grid, dim_block>>>(this->_size, 1, this->data, x.data_ptr(), all_eq_gpu.data_ptr());
497 
498  all_eq_gpu.get_val(&all_eq);
499  this->c->check();
500 
501  return (bool) all_eq;
502 }
