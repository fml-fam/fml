50  template <typename REAL>
51  void invert(gpumat<REAL> &x)
52  {
53  if (!x.is_square())
54  throw std::runtime_error("'x' must be a square matrix");
55 
56  // Factor x = LU
57  auto c = x.get_card();
58  gpuvec<int> p(c);
59  int info;
60  lu(x, p, info);
61  linalgutils::check_info(info, "getrf");
62 
63  // Invert
64  const len_t n = x.nrows();
65  const len_t nrhs = n;
66  gpumat<REAL> inv(c, n, nrhs);
67  inv.fill_eye();
68 
69  gpuscalar<int> info_device(c, info);
70 
71  gpulapack_status_t check = gpulapack::getrs(c->lapack_handle(), GPUBLAS_OP_N, n,
72  nrhs, x.data_ptr(), n, p.data_ptr(), inv.data_ptr(), n, info_device.data_ptr());
73 
74  info_device.get_val(&info);
75  gpulapack::err::check_ret(check, "getrs");
76  fml::linalgutils::check_info(info, "getrs");
77 
78  copy::gpu2gpu(inv, x);
79  }

102  template <typename REAL>
103  void trinv(const bool upper, const bool unit_diag, gpumat<REAL> &x)
104  {
105  if (!x.is_square())
106  throw std::runtime_error("'x' must be a square matrix");
107 
108  const len_t n = x.nrows();
109  gpumat<REAL> inv(x.get_card(), n, n);
110  inv.fill_eye();
111 
112  gpublas_fillmode_t uplo = (upper ? GPUBLAS_FILL_U : GPUBLAS_FILL_L);
113  gpublas_diagtype_t diag = (unit_diag ? GPUBLAS_DIAG_UNIT : GPUBLAS_DIAG_NON_UNIT);
114 
115  gpublas_status_t check = gpublas::trsm(x.get_card()->blas_handle(),
116  GPUBLAS_SIDE_LEFT, uplo, GPUBLAS_OP_N, diag, n, n, (REAL)1, x.data_ptr(),
117  n, inv.data_ptr(), n);
118 
119  gpublas::err::check_ret(check, "trsm");
120  copy::gpu2gpu(inv, x);
121 
122  char cuplo = (upper ? 'L' : 'U');
123  gpu_utils::tri2zero(cuplo, false, n, n, x.data_ptr(), n);
124  }
