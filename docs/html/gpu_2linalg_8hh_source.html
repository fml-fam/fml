335  template <typename REAL>
336  void lu(gpumat<REAL> &x, gpuvec<int> &p, int &info)
337  {
338  err::check_card(x, p);
339 
340  info = 0;
341  const len_t m = x.nrows();
342  const len_t n = x.ncols();
343  auto c = x.get_card();
344 
345  const len_t lipiv = std::min(m, n);
346  if (!p.get_card()->valid_card())
347  p.inherit(c);
348 
349  p.resize(lipiv);
350 
351  #if defined(FML_GPULAPACK_VENDOR)
352  int lwork;
353  gpulapack_status_t check = gpulapack::getrf_buflen(c->lapack_handle(), m,
354  n, x.data_ptr(), m, &lwork);
355  gpulapack::err::check_ret(check, "getrf_bufferSize");
356 
357  gpuvec<REAL> work(c, lwork);
358  gpuscalar<int> info_device(c, info);
359 
360  check = gpulapack::getrf(c->lapack_handle(), m, n, x.data_ptr(), m,
361  work.data_ptr(), p.data_ptr(), info_device.data_ptr());
362 
363  info_device.get_val(&info);
364  gpulapack::err::check_ret(check, "getrf");
365  #elif defined(FML_GPULAPACK_MAGMA)
366  cpuvec<int> p_cpu(lipiv);
367  gpulapack::getrf(m, n, x.data_ptr(), m, p_cpu.data_ptr(), &info);
368  gpuhelpers::cpu2gpu(p_cpu, p);
369  #else
370  #error "Unsupported GPU lapack"
371  #endif

390  static __global__ void kernel_lu_pivot_sgn(const len_t n, int *ipiv, int *sgn)
391  {
392  int i = blockDim.x*blockIdx.x + threadIdx.x;
393 
394  if (i < n)
395  {
396  ipiv[i] = (ipiv[i] != (i+1) ? -1 : 1);
397  atomicAdd(sgn, ipiv[i]);
398 
399  if (threadIdx.x == 0)
400  (*sgn) = ((*sgn)%2 == 0 ? 1 : -1);
401  }
402  }
403 
404  template <typename REAL>
405  __global__ void kernel_det_mod(const len_t m, const len_t n, const REAL *x, REAL *mod, int *sgn)
406  {
407  int i = blockDim.x*blockIdx.x + threadIdx.x;
408  int j = blockDim.y*blockIdx.y + threadIdx.y;
409 
410  if (i < m && j < n && i == j)
411  {
412  REAL d = x[i + m*j];
413  int s = 0;
414 
415  if (d < 0)
416  {
417  d = (*mod) + log(-d);
418  s++;
419  }
420  else
421  d = (*mod) + log(d);
422 
423  atomicAdd(mod, d);
424  int s_g = (*sgn);
425  (*sgn) = 0;
426  atomicAdd(sgn, s);
427 
428  if (threadIdx.x == 0 && threadIdx.y == 0)
429  {
430  (*sgn) = ((*sgn)%2 == 0 ? 1 : -1);
431  (*sgn) *= s_g;
432  }
433  }
434  }

455  template <typename REAL>
456  void det(gpumat<REAL> &x, int &sign, REAL &modulus)
457  {
458  if (!x.is_square())
459  throw std::runtime_error("'x' must be a square matrix");
460 
461  auto c = x.get_card();
462 
463  gpuvec<int> p(c);
464  int info;
465  lu(x, p, info);
466 
467  if (info != 0)
468  {
469  if (info > 0)
470  {
471  sign = 1;
472  modulus = -INFINITY;
473  return;
474  }
475  else
476  return;
477  }
478 
479  // get determinant
480  modulus = 0.0;
481  sign = 1;
482 
483  gpuscalar<int> sign_gpu(c, sign);
484  gpuscalar<REAL> modulus_gpu(c, modulus);
485 
486  kernel_lu_pivot_sgn<<<p.get_griddim(), p.get_blockdim()>>>(p.size(),
487  p.data_ptr(), sign_gpu.data_ptr());
488  kernel_det_mod<<<x.get_griddim(), x.get_blockdim()>>>(x.nrows(), x.ncols(),
489  x.data_ptr(), modulus_gpu.data_ptr(), sign_gpu.data_ptr());
490 
491  sign_gpu.get_val(&sign);
492  modulus_gpu.get_val(&modulus);
493  }
