407  int svd_internals(const int nu, const int nv, gpumat<REAL> &x, gpuvec<REAL> &s, gpumat<REAL> &u, gpumat<REAL> &vt)

435  gpulapack::err::check_ret(check, "gesvd_bufferSize");

438  gpuvec<REAL> rwork(c, minmn-1);
439 
440  int info = 0;

442 
443  check = gpulapack::gesvd(c->lapack_handle(), jobu, jobvt, m, n, x.data_ptr(),
444  m, s.data_ptr(), u.data_ptr(), m, vt.data_ptr(), minmn, work.data_ptr(),

500  int eig_sym_internals(const bool only_values, gpumat<REAL> &x,
501  gpuvec<REAL> &values, gpumat<REAL> &vectors)
502  {
503  if (!x.is_square())
504  throw std::runtime_error("'x' must be a square matrix");
505 
506  auto c = x.get_card();
507 
508  len_t n = x.nrows();
509  values.resize(c, n);
510 
511  cusolverEigMode_t jobz;
512  if (only_values)
513  jobz = CUSOLVER_EIG_MODE_NOVECTOR;
514  else
515  jobz = CUSOLVER_EIG_MODE_VECTOR;
516 
517  int lwork;
518  gpulapack_status_t check = gpulapack::syevd_buflen(c->lapack_handle(), jobz,
519  GPUBLAS_FILL_L, n, x.data_ptr(), n, values.data_ptr(), &lwork);
520  gpulapack::err::check_ret(check, "syevd_bufferSize");
521 
522  gpuvec<REAL> work(c, lwork);
523 
524  int info = 0;
525  gpuscalar<int> info_device(c, info);
526 
527  check = gpulapack::syevd(c->lapack_handle(), jobz, GPUBLAS_FILL_L,
528  n, x.data_ptr(), n, values.data_ptr(), work.data_ptr(), lwork,
529  info_device.data_ptr());
530 
531  info_device.get_val(&info);
532  gpulapack::err::check_ret(check, "syevd");
533 
534  if (!only_values)
535  {
536  vectors.resize(c, n, n);
537  gpuhelpers::gpu2gpu(x, vectors);
538  }
539 
540  return info;
541  }

606  template <typename REAL>
607  void invert(gpumat<REAL> &x)
608  {
609  if (!x.is_square())
610  throw std::runtime_error("'x' must be a square matrix");
611 
612  // Factor x = LU
613  auto c = x.get_card();
614  gpuvec<int> p(c);
615  int info;
616  lu(x, p, info);
617  fml::linalgutils::check_info(info, "getrf");
618 
619  // Invert
620  const len_t n = x.nrows();
621  const len_t nrhs = n;
622  gpumat<REAL> inv(c, n, nrhs);
623  inv.fill_eye();
624 
625  gpuscalar<int> info_device(c, info);
626 
627  gpulapack_status_t check = gpulapack::getrs(c->lapack_handle(), GPUBLAS_OP_N, n,
628  nrhs, x.data_ptr(), n, p.data_ptr(), inv.data_ptr(), n, info_device.data_ptr());
629 
630  info_device.get_val(&info);
631  gpulapack::err::check_ret(check, "getrs");
632  fml::linalgutils::check_info(info, "getrs");
633 
634  gpuhelpers::gpu2gpu(inv, x);
635  }
