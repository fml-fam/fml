232 inline std::vector<int> fml::comm::jid(const int n) const
233 {
234  std::vector<int> ret;
235 
236  if (n > _size)
237  {
238  int local = n / _size;
239  int rem = n % _size;
240 
241  if (rem == 0 || (_rank < (_size - rem)))
242  {
243  ret.resize(local);
244  for (int i=0; i<local; i++)
245  ret[i] = i + (_rank*local);
246  }
247  else
248  {
249  ret.resize(local+1);
250  for (int i=0; i<=local; i++)
251  ret[i] = i + (_rank*(local+1)) - (_size - rem);
252  }
253  }
254  else
255  {
256  if (n > _rank)
257  {
258  ret.resize(1);
259  ret[0] = _rank;
260  }
261  else
262  ret.resize(0);
263  }
264 
265  return ret;
266 }

447 template <typename T>
448 inline MPI_Datatype fml::comm::mpi_type_lookup(const T *x) const
449 {
450  (void) x;
451 
452  // C types
453  if (typeid(T) == typeid(char))
454  return MPI_CHAR;
455  else if (typeid(T) == typeid(double))
456  return MPI_DOUBLE;
457  else if (typeid(T) == typeid(float))
458  return MPI_FLOAT;
459  else if (typeid(T) == typeid(int))
460  return MPI_INT;
461  else if (typeid(T) == typeid(long))
462  return MPI_LONG;
463  else if (typeid(T) == typeid(long double))
464  return MPI_LONG_DOUBLE;
465  else if (typeid(T) == typeid(long long))
466  return MPI_LONG_LONG_INT;
467  else if (typeid(T) == typeid(short))
468  return MPI_SHORT;
469  else if (typeid(T) == typeid(unsigned int))
470  return MPI_UNSIGNED;
471  else if (typeid(T) == typeid(unsigned char))
472  return MPI_UNSIGNED_CHAR;
473  else if (typeid(T) == typeid(unsigned long))
474  return MPI_UNSIGNED_LONG;
475  else if (typeid(T) == typeid(unsigned short))
476  return MPI_UNSIGNED_SHORT;
477  else if (typeid(T) == typeid(uint32_t))
478  return MPI_UINT32_T;
479 
480  // stdint types
481  else if (typeid(T) == typeid(int8_t))
482  return MPI_INT8_T;
483  else if (typeid(T) == typeid(int16_t))
484  return MPI_INT16_T;
485  else if (typeid(T) == typeid(int32_t))
486  return MPI_INT32_T;
487  else if (typeid(T) == typeid(int64_t))
488  return MPI_INT64_T;
489  else if (typeid(T) == typeid(uint8_t))
490  return MPI_UINT8_T;
491  else if (typeid(T) == typeid(uint16_t))
492  return MPI_UINT16_T;
493  else if (typeid(T) == typeid(uint32_t))
494  return MPI_UINT32_T;
495  else if (typeid(T) == typeid(uint64_t))
496  return MPI_UINT64_T;
497 
498  else
499  return MPI_DATATYPE_NULL;
500 }
