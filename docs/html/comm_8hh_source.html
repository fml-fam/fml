229 inline std::vector<int> comm::jid(const int n) const
230 {
231  std::vector<int> ret;
232 
233  if (n > _size)
234  {
235  int local = n / _size;
236  int rem = n % _size;
237 
238  if (rem == 0 || (_rank < (_size - rem)))
239  {
240  ret.resize(local);
241  for (int i=0; i<local; i++)
242  ret[i] = i + (_rank*local);
243  }
244  else
245  {
246  ret.resize(local+1);
247  for (int i=0; i<=local; i++)
248  ret[i] = i + (_rank*(local+1)) - (_size - rem);
249  }
250  }
251  else
252  {
253  if (n > _rank)
254  {
255  ret.resize(1);
256  ret[0] = _rank;
257  }
258  else
259  ret.resize(0);
260  }
261 
262  return ret;
263 }

444 template <typename T>
445 inline MPI_Datatype comm::mpi_type_lookup(const T *x) const
446 {
447  (void) x;
448 
449  // C types
450  if (typeid(T) == typeid(char))
451  return MPI_CHAR;
452  else if (typeid(T) == typeid(double))
453  return MPI_DOUBLE;
454  else if (typeid(T) == typeid(float))
455  return MPI_FLOAT;
456  else if (typeid(T) == typeid(int))
457  return MPI_INT;
458  else if (typeid(T) == typeid(long))
459  return MPI_LONG;
460  else if (typeid(T) == typeid(long double))
461  return MPI_LONG_DOUBLE;
462  else if (typeid(T) == typeid(long long))
463  return MPI_LONG_LONG_INT;
464  else if (typeid(T) == typeid(short))
465  return MPI_SHORT;
466  else if (typeid(T) == typeid(unsigned int))
467  return MPI_UNSIGNED;
468  else if (typeid(T) == typeid(unsigned char))
469  return MPI_UNSIGNED_CHAR;
470  else if (typeid(T) == typeid(unsigned long))
471  return MPI_UNSIGNED_LONG;
472  else if (typeid(T) == typeid(unsigned short))
473  return MPI_UNSIGNED_SHORT;
474  else if (typeid(T) == typeid(uint32_t))
475  return MPI_UINT32_T;
476 
477  // stdint types
478  else if (typeid(T) == typeid(int8_t))
479  return MPI_INT8_T;
480  else if (typeid(T) == typeid(int16_t))
481  return MPI_INT16_T;
482  else if (typeid(T) == typeid(int32_t))
483  return MPI_INT32_T;
484  else if (typeid(T) == typeid(int64_t))
485  return MPI_INT64_T;
486  else if (typeid(T) == typeid(uint8_t))
487  return MPI_UINT8_T;
488  else if (typeid(T) == typeid(uint16_t))
489  return MPI_UINT16_T;
490  else if (typeid(T) == typeid(uint32_t))
491  return MPI_UINT32_T;
492  else if (typeid(T) == typeid(uint64_t))
493  return MPI_UINT64_T;
494 
495  else
496  return MPI_DATATYPE_NULL;
497 }
