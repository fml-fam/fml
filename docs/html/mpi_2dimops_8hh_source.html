28  template <typename REAL>
29  void rowsums(const mpimat<REAL> &x, cpuvec<REAL> &s)
30  {
31  const REAL *x_d = x.data_ptr();
32  const len_t m = x.nrows();
33  const len_t m_local = x.nrows_local();
34  const len_t n_local = x.ncols_local();
35  const len_t mb = x.bf_rows();
36 
37  const grid g = x.get_grid();
38 
39  s.resize(m);
40  s.fill_zero();
41  REAL *s_d = s.data_ptr();
42 
43  for (len_t j=0; j<n_local; j++)
44  {
45  #pragma omp for simd
46  for (len_t i=0; i<m_local; i++)
47  {
48  const len_t gi = fml::bcutils::l2g(i, mb, g.nprow(), g.myrow());
49  s_d[gi] += x_d[i + m_local*j];
50  }
51  }
52 
53  x.get_grid().allreduce(m, 1, s_d, 'A');
54  }

129  static inline void col_mean(const grid g, const len_t j, const len_t m, const len_t m_local, const REAL *x, REAL &mean)
130  {
131  mean = 0;
132  fml::vecops::cpu::sum(m_local, x + m_local*j, mean);
133  g.allreduce(1, 1, &mean, 'C');
134  mean /= (REAL) m;
135  }

137  template <typename REAL>
138  static inline void col_var(const grid g, const len_t j, const len_t m, const len_t m_local, const REAL *x, const REAL &mean, REAL *work, REAL &var)
139  {
140  work[0] = 0;
141  work[1] = 0;
142 
143  for (len_t i = 0; i<m_local; i++)
144  {
145  REAL diff = x[i + m_local*j] - mean;
146  work[0] += diff*diff;
147  work[1] += diff;
148  }
149 
150  g.allreduce(2, 1, work, 'C');
151 
152  var = (work[0] - work[1]*work[1]/m) / (m-1);
153  }

175  template <typename REAL>
176  static inline void scale(mpimat<REAL> &x)
177  {
178  REAL *x_d = x.data_ptr();
179  const len_t m = x.nrows();
180  const len_t m_local = x.nrows_local();
181  const len_t n_local = x.ncols_local();
182 
183  grid g = x.get_grid();
184 
185  REAL work[2];
186 
187  for (len_t j=0; j<n_local; j++)
188  {
189  REAL mean;
190  col_mean(g, j, m, m_local, x_d, mean);
191 
192  REAL var;
193  col_var(g, j, m, m_local, x_d, mean, work, var);
194  var = (REAL)1.0/sqrt(var);
195  fml::vecops::cpu::sweep_mul(var, m_local, x_d + m_local*j);
196  }
197  }

201  {

204  const len_t m_local = x.nrows_local();
205  const len_t n_local = x.ncols_local();
206 
207  grid g = x.get_grid();
208 
209  REAL work[2];
210 
211  for (len_t j=0; j<n_local; j++)
212  {
213  REAL mean;
214  col_mean(g, j, m, m_local, x_d, mean);
215 
216  REAL var;
217  col_var(g, j, m, m_local, x_d, mean, work, var);
218 
219  #pragma omp for simd
220  for (len_t i=0; i<m_local; i++)
221  x_d[i + m_local*j] = (x_d[i + m_local*j] - mean) / sqrt(var);
222  }
223  }
