30  template <typename REAL>
31  void rowsums(const mpimat<REAL> &x, cpuvec<REAL> &s)
32  {
33  const REAL *x_d = x.data_ptr();
34  const len_t m = x.nrows();
35  const len_t m_local = x.nrows_local();
36  const len_t n_local = x.ncols_local();
37  const len_t mb = x.bf_rows();
38 
39  const grid g = x.get_grid();
40 
41  s.resize(m);
42  s.fill_zero();
43  REAL *s_d = s.data_ptr();
44 
45  for (len_t j=0; j<n_local; j++)
46  {
47  for (len_t i=0; i<m_local; i++)
48  {
49  const len_t gi = fml::bcutils::l2g(i, mb, g.nprow(), g.myrow());
50  s_d[gi] += x_d[i + m_local*j];
51  }
52  }
53 
54  x.get_grid().allreduce(m, 1, s_d, 'A');
55  }

85  void colsums(const mpimat<REAL> &x, cpuvec<REAL> &s)
86  {
87  const REAL *x_d = x.data_ptr();
88  const len_t n = x.ncols();
89  const len_t m_local = x.nrows_local();
90  const len_t n_local = x.ncols_local();
91  const len_t nb = x.bf_cols();
92 
93  const grid g = x.get_grid();
94 
95  s.resize(n);
96  s.fill_zero();
97  REAL *s_d = s.data_ptr();
98 
99  #pragma omp parallel for if(m_local*n_local > fml::omp::OMP_MIN_SIZE)
100  for (len_t j=0; j<n_local; j++)
101  {
102  const len_t gj = fml::bcutils::l2g(j, nb, g.npcol(), g.mycol());
103  fml::vecops::cpu::sum(m_local, x_d + m_local*j, s_d[gj]);
104  }
105 
106  x.get_grid().allreduce(n, 1, s_d, 'A');

130  template <typename REAL>
131  static inline void col_mean(const grid g, const len_t j, const len_t m, const len_t m_local, const REAL *x, REAL &mean)
132  {
133  mean = 0;
134  fml::vecops::cpu::sum(m_local, x + m_local*j, mean);
135  g.allreduce(1, 1, &mean, 'C');
136  mean /= (REAL) m;
137  }

139  template <typename REAL>
140  static inline void col_var(const grid g, const len_t j, const len_t m, const len_t m_local, const REAL *x, const REAL &mean, REAL *work, REAL &var)
141  {
142  work[0] = 0;
143  work[1] = 0;
144 
145  for (len_t i = 0; i<m_local; i++)
146  {
147  REAL diff = x[i + m_local*j] - mean;
148  work[0] += diff*diff;
149  work[1] += diff;
150  }
151 
152  g.allreduce(2, 1, work, 'C');
153 
154  var = (work[0] - work[1]*work[1]/m) / (m-1);
155  }

177  template <typename REAL>
178  static inline void scale(mpimat<REAL> &x)
179  {
180  REAL *x_d = x.data_ptr();
181  const len_t m = x.nrows();
182  const len_t m_local = x.nrows_local();
183  const len_t n_local = x.ncols_local();
184 
185  grid g = x.get_grid();
186 
187  REAL work[2];
188 
189  for (len_t j=0; j<n_local; j++)
190  {
191  REAL mean;
192  col_mean(g, j, m, m_local, x_d, mean);
193 
194  REAL var;
195  col_var(g, j, m, m_local, x_d, mean, work, var);
196  var = (REAL)1.0/sqrt(var);
197  fml::vecops::cpu::sweep_mul(var, m_local, x_d + m_local*j);
198  }
199  }

208

210

212 
213  for (len_t j=0; j<n_local; j++)
214  {
215  REAL mean;
216  col_mean(g, j, m, m_local, x_d, mean);
217 
218  REAL var;
219  col_var(g, j, m, m_local, x_d, mean, work, var);
220 
221  #pragma omp simd
222  for (len_t i=0; i<m_local; i++)
223  x_d[i + m_local*j] = (x_d[i + m_local*j] - mean) / sqrt(var);
224  }
