229 template <typename REAL>
230 void fml::cpumat<REAL>::resize(len_t nrows, len_t ncols)
231 {
232  check_params(nrows, ncols);
233 
234  const size_t len = (size_t) nrows * ncols * sizeof(REAL);
235  const size_t oldlen = (size_t) this->m * this->n * sizeof(REAL);
236 
237  if ((nrows == 0 || ncols == 0) || (len == oldlen))
238  {
239  this->m = nrows;
240  this->n = ncols;
241  return;
242  }
243 
244  void *realloc_ptr;
245  if (oldlen == 0)
246  realloc_ptr = malloc(len);
247  else
248  realloc_ptr = realloc(this->data, len);
249 
250  if (realloc_ptr == NULL)
251  throw std::bad_alloc();
252 
253  this->data = (REAL*) realloc_ptr;
254 
255  this->m = nrows;
256  this->n = ncols;
257 }

391 template <typename REAL>
392 void fml::cpumat<REAL>::fill_linspace(const REAL start, const REAL stop)
393 {
394  if (start == stop)
395  this->fill_val(start);
396  else
397  {
398  const REAL v = (stop-start)/((REAL) this->m*this->n - 1);
399 
400  #pragma omp parallel for if((this->m)*(this->n) > fml::omp::OMP_MIN_SIZE)
401  for (len_t j=0; j<this->n; j++)
402  {
403  #pragma omp simd
404  for (len_t i=0; i<this->m; i++)
405  {
406  const len_t ind = i + this->m*j;
407  this->data[ind] = v*((REAL) ind) + start;
408  }
409  }
410  }

819  return false;
