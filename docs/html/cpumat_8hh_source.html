200 template <typename REAL>
201 void cpumat<REAL>::resize(len_t nrows, len_t ncols)
202 {
203  check_params(nrows, ncols);
204 
205  const size_t len = (size_t) nrows * ncols * sizeof(REAL);
206  const size_t oldlen = (size_t) this->m * this->n * sizeof(REAL);
207 
208  if ((nrows == 0 || ncols == 0) || (len == oldlen))
209  {
210  this->m = nrows;
211  this->n = ncols;
212  return;
213  }
214 
215  void *realloc_ptr;
216  if (oldlen == 0)
217  realloc_ptr = malloc(len);
218  else
219  realloc_ptr = realloc(this->data, len);
220 
221  if (realloc_ptr == NULL)
222  throw std::bad_alloc();
223 
224  this->data = (REAL*) realloc_ptr;
225 
226  this->m = nrows;
227  this->n = ncols;
228 }

347 template <typename REAL>
348 void cpumat<REAL>::fill_linspace(const REAL start, const REAL stop)
349 {
350  if (start == stop)
351  this->fill_val(start);
352  else
353  {
354  const REAL v = (stop-start)/((REAL) this->m*this->n - 1);
355 
356  #pragma omp parallel for if((this->m)*(this->n) > fml::omp::OMP_MIN_SIZE)
357  for (len_t j=0; j<this->n; j++)
358  {
359  #pragma omp simd
360  for (len_t i=0; i<this->m; i++)
361  {
362  const len_t ind = i + this->m*j;
363  this->data[ind] = v*((REAL) ind) + start;
364  }
365  }
366  }
