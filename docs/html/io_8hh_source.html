80  void read_cpu_chunk(const char *filename, const len_t row_first, const len_t row_last, cpumat<REAL> &x)
81  {
82  FILE *f;
83  f = fopen(filename, "rb");
84  if (!f)
85  throw std::runtime_error("unable to open file for reading");
86 
87  char tmp[HEADER_LEN];
88  int nread = fread(tmp, sizeof(char), HEADER_LEN, f);

90  throw std::runtime_error("mal-formed file");
91 
92  len_t m, n;
93  nread = fread(&m, sizeof(len_t), 1, f);
94  nread = fread(&n, sizeof(len_t), 1, f);
95 
96  if (row_first > row_last || row_first < 0 || row_last < 0 || row_first >= m || row_last >= m)
97  throw std::runtime_error("bad first/last row");
98 
99  const len_t m_chunk = row_last - row_first + 1;
100  x.resize(m_chunk, n);
101 
102  for (len_t j=0; j<n; j++)
103  {
104  fseek(f, (row_first + m*j)*sizeof(REAL) + read_offset(), SEEK_SET);
105  nread = fread(x.data_ptr() + m_chunk*j, sizeof(REAL), m_chunk, f);
106  }
107 
108  fclose(f);
109  }
110 }
