371  template <typename REAL>
372  REAL trace(const mpimat<REAL> &x)
373  {
374  const REAL *x_d = x.data_ptr();
375  const len_t minmn = std::min(x.nrows(), x.ncols());
376  const len_t m_local = x.nrows_local();
377  const int mb = x.bf_rows();
378  const int nb = x.bf_cols();
379  grid g = x.get_grid();
380 
381  REAL tr = 0;
382  for (len_t gi=0; gi<minmn; gi++)
383  {
384  const len_local_t i = fml::bcutils::g2l(gi, mb, g.nprow());
385  const len_local_t j = fml::bcutils::g2l(gi, nb, g.npcol());
386 
387  const int pr = fml::bcutils::g2p(gi, mb, g.nprow());
388  const int pc = fml::bcutils::g2p(gi, nb, g.npcol());
389 
390  if (pr == g.myrow() && pc == g.mycol())
391  tr += x_d[i + m_local*j];
392  }
393 
394  g.allreduce(1, 1, &tr, 'A');
395 
396  return tr;
397  }
