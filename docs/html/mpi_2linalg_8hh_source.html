383  template <typename REAL>
384  void det(mpimat<REAL> &x, int &sign, REAL &modulus)
385  {
386  if (!x.is_square())
387  throw std::runtime_error("'x' must be a square matrix");
388 
389  cpuvec<int> p;
390  int info;
391  lu(x, p, info);
392 
393  if (info != 0)
394  {
395  if (info > 0)
396  {
397  sign = 1;
398  modulus = -INFINITY;
399  return;
400  }
401  else
402  return;
403  }
404 
405 
406  // get determinant
407  REAL mod = 0.0;
408  int sgn = 1;
409 
410  const len_t m_local = x.nrows_local();
411  const len_t n_local = x.ncols_local();
412 
413  const int *ipiv = p.data_ptr();
414  const REAL *a = x.data_ptr();
415  const grid g = x.get_grid();
416 
417  for (len_t i=0; i<m_local; i++)
418  {
419  len_t gi = fml::bcutils::l2g(i, x.bf_rows(), g.nprow(), g.myrow());
420 
421  if (ipiv[i] != (gi + 1))
422  sgn = -sgn;
423  }
424 
425  for (len_t j=0; j<n_local; j++)
426  {
427  for (len_t i=0; i<m_local; i++)
428  {
429  len_t gi = fml::bcutils::l2g(i, x.bf_rows(), g.nprow(), g.myrow());
430  len_t gj = fml::bcutils::l2g(j, x.bf_cols(), g.npcol(), g.mycol());
431 
432  if (gi == gj)
433  {
434  const REAL d = a[i + m_local*j];
435  if (d < 0)
436  {
437  mod += log(-d);
438  sgn *= -1;
439  }
440  else
441  mod += log(d);
442  }
443  }
444  }
445 
446  g.allreduce(1, 1, &mod);
447 
448  sgn = (sgn<0 ? 1 : 0);
449  g.allreduce(1, 1, &sgn, 'C');
450  sgn = (sgn%2==0 ? 1 : -1);
451 
452  modulus = mod;
453  sign = sgn;
454  }

467  template <typename REAL>
468  REAL trace(const mpimat<REAL> &x)
469  {
470  const REAL *x_d = x.data_ptr();
471  const len_t minmn = std::min(x.nrows(), x.ncols());
472  const len_t m_local = x.nrows_local();
473  const int mb = x.bf_rows();
474  const int nb = x.bf_cols();
475  const grid g = x.get_grid();
476 
477  REAL tr = 0;
478  for (len_t gi=0; gi<minmn; gi++)
479  {
480  const len_local_t i = fml::bcutils::g2l(gi, mb, g.nprow());
481  const len_local_t j = fml::bcutils::g2l(gi, nb, g.npcol());
482 
483  const int pr = fml::bcutils::g2p(gi, mb, g.nprow());
484  const int pc = fml::bcutils::g2p(gi, nb, g.npcol());
485 
486  if (pr == g.myrow() && pc == g.mycol())
487  tr += x_d[i + m_local*j];
488  }
489 
490  g.allreduce(1, 1, &tr, 'A');
491 
492  return tr;
493  }

532
