15  template <typename REAL>
16  void det(cpumat<REAL> &x, int &sign, REAL &modulus)
17  {
18  const len_t m = x.nrows();
19  if (!x.is_square())
20  throw std::runtime_error("'x' must be a square matrix");
21 
22  cpuvec<int> p;
23  int info = lu(x, p);
24 
25  if (info != 0)
26  {
27  if (info > 0)
28  {
29  sign = 1;
30  modulus = -INFINITY;
31  return;
32  }
33  else
34  return;
35  }
36 
37 
38  // get determinant
39  REAL mod = 0.0;
40  int sgn = 1;
41 
42  const int *ipiv = p.data_ptr();
43  for (int i=0; i<m; i++)
44  {
45  if (ipiv[i] != (i + 1))
46  sgn = -sgn;
47  }
48 
49  const REAL *a = x.data_ptr();
50 
51  #pragma omp parallel for reduction(+:mod) reduction(*:sgn)
52  for (int i=0; i<m; i+=m+1)
53  {
54  const REAL d = a[i + m*i];
55  if (d < 0)
56  {
57  mod += log(-d);
58  sgn *= -1;
59  }
60  else
61  mod += log(d);
62  }
63 
64  modulus = mod;
65  sign = sgn;
66  }

73  void qr(cpumat<REAL> &x, cpuvec<REAL> &qraux)
74  {
75  len_t m = x.nrows();
76  len_t n = x.ncols();
77  len_t minmn = std::min(m, n);
78 
79  qraux.resize(minmn);
80 
81  REAL tmp;
82  geqp3(m, n, 0.f, m, 0, 0.f, &tmp, -1, 0);
83  int lwork = std::max((int) tmp, 1);
84  cpuvec<REAL> work(lwork);
85 
86  cpuvec<int> pivot(n);
87  pivot.fill_zero();
88 
89  int info = 0;
90  geqp3(m, n, x.data_ptr(), m, pivot.data_ptr(), qraux.data_ptr(), work.data_ptr(), lwork, &info);
91 
92  if (info != 0)
93  linalgutils::check_info(info, "geqp3");
94  }

96  template <typename REAL>
97  void q(const cpumat<REAL> &QR, const cpuvec<REAL> &qraux, cpumat<REAL> &Q)
98  {
99  len_t m = QR.nrows();
100  len_t n = QR.ncols();
101 
102  len_t nrhs = std::min(m, n);
103  Q.resize(m, nrhs);
104  Q.fill_zero();
105 
106  REAL *Q_d = Q.data_ptr();
107  for (len_t i=0; i<m*nrhs; i+=m+1)
108  Q_d[i] = 1.f;
109 
110  int info = 0;
111  float tmp;
112  int lwork = -1;
113  ormqr('L', 'N', m, nrhs, n, QR.data_ptr(), m, qraux.data_ptr(), Q_d, m, &tmp, lwork, &info);
114  lwork = (int) tmp;
115  cpumat<REAL> work(lwork);
116 
117  ormqr('L', 'N', m, nrhs, n, QR.data_ptr(), m, qraux.data_ptr(), Q_d, m, work.data_ptr(), lwork, &info);
118 
119  if (info != 0)
120  linalgutils::check_info(info, "ormqr");
121  }
