172 inline grid::grid(const gridshape gridtype)
173 {
174  char order = 'R';
175 
176  int mypnum;
177  Cblacs_pinfo(&mypnum, &_nprocs);
178 
179  Cblacs_get(-1, 0, &_ictxt);
180 
181  if (gridtype == PROC_GRID_SQUARE)
182  {
183  int nr, nc;
184  squarish(&nr, &nc);
185 
186  Cblacs_gridinit(&_ictxt, &order, nr, nc);
187  }
188  else if (gridtype == PROC_GRID_TALL)
189  Cblacs_gridinit(&_ictxt, &order, _nprocs, 1);
190  else if (gridtype == PROC_GRID_WIDE)
191  Cblacs_gridinit(&_ictxt, &order, 1, _nprocs);
192  else
193  throw std::runtime_error("Process grid should be one of PROC_GRID_SQUARE, PROC_GRID_TALL, or PROC_GRID_WIDE");
194 
195  Cblacs_gridinfo(_ictxt, &_nprow, &_npcol, &_myrow, &_mycol);
196 }

526 inline void grid::squarish(int *nr, int *nc) const
527 {
528  int n = (int) sqrt((double) _nprocs);
529  n = (n<1)?1:n; // suppresses bogus compiler warning
530 
531  for (int i=0; i<n; i++)
532  {
533  (*nc) = n - i;
534  (*nr) = _nprocs % (*nc);
535  if ((*nr) == 0)
536  break;
537  }
538 
539  (*nr) = _nprocs / (*nc);
540 }
