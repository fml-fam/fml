208 template <typename T>
209 void fml::cpuvec<T>::resize(len_t size)
210 {
211  check_params(size);
212 
213  if (size == 0)
214  {
215  this->_size = size;
216  return;
217  }
218  else if (this->_size == size)
219  return;
220 
221  const size_t len = (size_t) size * sizeof(T);
222 
223  void *realloc_ptr;
224  if (this->_size == 0)
225  realloc_ptr = malloc(len);
226  else
227  realloc_ptr = realloc(this->data, len);
228 
229  if (realloc_ptr == NULL)
230  throw std::bad_alloc();
231 
232  this->data = (T*) realloc_ptr;
233  this->_size = size;
234 }

344 template <typename REAL>
345 void fml::cpuvec<REAL>::fill_linspace(const REAL start, const REAL stop)
346 {
347  if (start == stop)
348  this->fill_val(start);
349  else
350  {
351  const REAL v = (stop-start)/((REAL) this->_size - 1);
352

376 template <typename REAL>
377 void fml::cpuvec<REAL>::subset(const len_t start, const len_t stop, const bool interior)
378 {
379  len_t size_new = interior ? stop-start : this->_size - (stop-start);
380  size_t len = size_new * sizeof(REAL);
381  REAL *data_new = (REAL*) malloc(len);
382  if (data_new == NULL)
383  throw std::bad_alloc();
384 
385  if (interior)
386  std::memcpy(data_new, this->data + start, len);
387  else
388  {
389  len_t n = std::max(start-1, 0);
390  std::memcpy(data_new, this->data, n*sizeof(REAL));
391  if (stop < this->_size)
392  {
393  n = this->_size-stop;
394  std::memcpy(data_new, this->data + stop, n*sizeof(REAL));
395  }
396  }
397 
398  std::free(this->data);
399  this->data = data_new;
400  this->_size = size_new;
401 }
