34  void tssvd(gpumat<REAL> &x, gpuvec<REAL> &s, gpumat<REAL> &u, gpumat<REAL> &vt)
35  {
36  const len_t m = x.nrows();
37  const len_t n = x.ncols();
38  if (m <= n)
39  throw std::runtime_error("'x' must have more rows than cols");
40 
41  auto c = x.get_card();
42 
43  gpuvec<REAL> qraux(c);
44  gpuvec<REAL> work(c);
45  qr_internals(false, x, qraux, work);
46 
47  gpumat<REAL> R(c, n, n);
48  qr_R(x, R);
49 
50  gpumat<REAL> u_R(c, n, n);
51  int info = svd_internals(1, 1, R, s, u_R, vt);
52  fml::linalgutils::check_info(info, "gesvd");
53 
54  u.resize(m, n);
55  qr_Q(x, qraux, u, work);
56 
57  matmult(false, false, (REAL)1.0, u, u_R, x);
58  copy::gpu2gpu(x, u);
59  }

61  template <typename REAL>
62  void tssvd(gpumat<REAL> &x, gpuvec<REAL> &s)
63  {
64  const len_t m = x.nrows();
65  const len_t n = x.ncols();
66  if (m <= n)
67  throw std::runtime_error("'x' must have more rows than cols");
68 
69  auto c = x.get_card();
70  s.resize(n);
71 
72  gpuvec<REAL> qraux(c);
73  gpuvec<REAL> work(c);
74  qr_internals(false, x, qraux, work);
75 
76  fml::gpu_utils::tri2zero('L', false, n, n, x.data_ptr(), m);
77 
78  int lwork;
79  gpulapack_status_t check = gpulapack::gesvd_buflen(c->lapack_handle(), n, n,
80  x.data_ptr(), &lwork);
81  gpulapack::err::check_ret(check, "gesvd_bufferSize");
82 
83  if (lwork > work.size())
84  work.resize(lwork);
85  if (m-1 > qraux.size())
86  qraux.resize(m-1);
87 
88  int info = 0;
89  gpuscalar<int> info_device(c, info);
90 
91  check = gpulapack::gesvd(c->lapack_handle(), 'N', 'N', n, n, x.data_ptr(),
92  m, s.data_ptr(), NULL, m, NULL, 1, work.data_ptr(), lwork,
93  qraux.data_ptr(), info_device.data_ptr());
94 
95  info_device.get_val(&info);
96  gpulapack::err::check_ret(check, "gesvd");
97  fml::linalgutils::check_info(info, "gesvd");
98  }

164  {

217  template <typename REAL>
218  void cpsvd(const gpumat<REAL> &x, gpuvec<REAL> &s, gpumat<REAL> &u, gpumat<REAL> &vt)
219  {
220  err::check_card(x, s);
221  err::check_card(x, u);
222  err::check_card(x, vt);
223 
224  const len_t m = x.nrows();
225  const len_t n = x.ncols();
226  const len_t minmn = std::min(m, n);
227 
228  auto c = x.get_card();
229 
230  gpumat<REAL> cp(c);
231 
232  if (m >= n)
233  {
234  crossprod((REAL)1.0, x, cp);
235  eigen_sym(cp, s, vt);
236  vt.rev_cols();
237  copy::gpu2gpu(vt, cp);
238  }
239  else
240  {
241  tcrossprod((REAL)1.0, x, cp);
242  eigen_sym(cp, s, u);
243  u.rev_cols();
244  copy::gpu2gpu(u, cp);
245  }
246 
247  s.rev();
248  auto sgrid = s.get_griddim();
249  auto sblock = s.get_blockdim();
250  fml::kernelfuns::kernel_root_abs<<<sgrid, sblock>>>(s.size(), s.data_ptr());
251 
252  REAL *ev_d;
253  if (m >= n)
254  ev_d = vt.data_ptr();
255  else
256  ev_d = cp.data_ptr();
257 
258  auto xgrid = x.get_griddim();
259  auto xblock = x.get_blockdim();
260  fml::kernelfuns::kernel_sweep_cols_div<<<xgrid, xblock>>>(minmn, minmn,
261  ev_d, s.data_ptr());
262 
263  if (m >= n)
264  {
265  matmult(false, false, (REAL)1.0, x, vt, u);
266  xpose(cp, vt);
267  }
268  else
269  matmult(true, false, (REAL)1.0, cp, x, vt);
270  }
