178 inline void card::info() const
179 {
180  nvml::init();
181 
182  int version = nvml::system::get_cuda_driver_version();
183  int version_major = version / 1000;
184  int version_minor = (version % 1000) / 10;
185 
186  nvmlDevice_t device = nvml::device::get_handle_by_index(_id);
187  std::string name = nvml::device::get_name(device);
188  double mem_used, mem_total;
189  nvml::device::get_memory_info(device, &mem_used, &mem_total);
190 
191  printf("## GPU %d ", _id);
192  printf("(%s) ", name.c_str());
193  printf("%.0f/%.0f MB ", mem_used/1024/1024, mem_total/1024/1024);
194  printf("- CUDA %d.%d", version_major, version_minor);
195  printf("\n\n");
196 
197  nvml::shutdown();
198 }

371 inline void card::init()
372 {
373  if (_id == UNINITIALIZED_CARD)
374  throw std::runtime_error("invalid card (uninitialized)");
375  else if (_id == DESTROYED_CARD)
376  throw std::runtime_error("invalid card (destroyed)");
377 
378  err = gpuprims::gpu_set_device(_id);
379  check_gpu_error();
380 }

384 inline void card::cleanup()
385 {
386  init();
387 
388  if (_lapack_handle)
389  {
390  gpuprims::gpu_lapack_free(_lapack_handle);
391  _lapack_handle = NULL;
392  }
393 
394  if (_blas_handle)
395  {
396  gpuprims::gpu_blas_free(_blas_handle);
397  _blas_handle = NULL;
398  }
399 
400  err = gpuprims::gpu_device_reset();
401 
402  _id = DESTROYED_CARD;
403 }

414  throw std::runtime_error(s);
