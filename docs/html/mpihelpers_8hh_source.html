45  template <typename REAL_IN, typename REAL_OUT>
46  void mpi2cpu_all(const mpimat<REAL_IN> &mpi, cpumat<REAL_OUT> &cpu)
47  {
48  grid g = mpi.get_grid();
49  if (!g.ingrid())
50  return;
51 
52  len_local_t m_local = mpi.nrows_local();
53  len_local_t n_local = mpi.ncols_local();
54 
55  int mb = mpi.bf_rows();
56 
57  len_t m = mpi.nrows();
58  len_t n = mpi.ncols();
59 
60  if (m != cpu.nrows() || n != cpu.ncols())
61  cpu.resize(m, n);
62 
63  cpu.fill_zero();
64 
65  REAL_OUT *gbl = cpu.data_ptr();
66  const REAL_IN *sub = mpi.data_ptr();
67 
68  if (m_local > 0 && n_local > 0)
69  {
70  for (len_local_t j=0; j<n_local; j++)
71  {
72  const int gj = fml::bcutils::l2g(j, mpi.bf_cols(), g.npcol(), g.mycol());
73 
74  for (len_local_t i=0; i<m_local; i+=mb)
75  {
76  const int gi = fml::bcutils::l2g(i, mpi.bf_rows(), g.nprow(), g.myrow());
77 
78  for (int ii=0; ii<mb && ii+i<m_local; ii++)
79  gbl[gi+ii + m*gj] = (REAL_OUT) sub[i+ii + m_local*j];
80  }
81  }
82  }
83 
84  g.allreduce(m, n, gbl);
85  }

125  template <typename REAL_IN, typename REAL_OUT>
126  void mpi2cpu(const mpimat<REAL_IN> &mpi, cpumat<REAL_OUT> &cpu, int rdest=0, int cdest=0)
127  {
128  grid g = mpi.get_grid();
129  if (!g.ingrid())
130  return;
131 
132  bool i_am_ret = (g.myrow() == rdest && g.mycol() == cdest) ? true : false;
133 
134  len_local_t m_local = mpi.nrows_local();
135 
136  int mb = mpi.bf_rows();
137  int nb = mpi.bf_cols();
138 
139  len_t m = mpi.nrows();
140  len_t n = mpi.ncols();
141 
142  if (i_am_ret)
143  {
144  if (m != cpu.nrows() || n != cpu.ncols())
145  cpu.resize(m, n);
146 
147  cpu.fill_zero();
148  }
149 
150  REAL_OUT *gbl = cpu.data_ptr();
151  const REAL_IN *sub = mpi.data_ptr();
152 
153  for (len_t gj=0; gj<n; gj+=nb)
154  {
155  const int pc = fml::bcutils::g2p(gj, nb, g.npcol());
156  const int j = fml::bcutils::g2l(gj, nb, g.npcol());
157 
158  for (len_t gi=0; gi<m; gi+=mb)
159  {
160  const int pr = fml::bcutils::g2p(gi, mb, g.nprow());
161  const int i = fml::bcutils::g2l(gi, mb, g.nprow());
162 
163  const int row_copylen = std::min(mb, m-gi);
164  const int col_copylen = std::min(nb, n-gj);
165 
166  if (i_am_ret)
167  {
168  if (pr == g.myrow() && pc == g.mycol())
169  {
170  for (int jj=0; jj<col_copylen; jj++)
171  arraytools::copy(row_copylen, sub + i+m_local*(j+jj), gbl + gi+m*(gj+jj));
172  }
173  else
174  g.recv(row_copylen, col_copylen, m, gbl + gi+m*j, pr, pc);
175  }
176  else if (pr == g.myrow() && pc == g.mycol())
177  g.send(row_copylen, col_copylen, m_local, sub + i+m_local*j, rdest, cdest);
178  }
179  }
180  }

244  {

246

248  }
249  }
250  }
