57  template <typename REAL_IN, typename REAL_OUT>
58  void copy_gpu2cpu(const len_t m, const len_t n, std::shared_ptr<card> c, const REAL_IN *in, REAL_OUT *out)
59  {
60  if (std::is_same<REAL_IN, REAL_OUT>::value)
61  {
62  const size_t len = (size_t) m*n*sizeof(REAL_IN);
63  c->mem_gpu2cpu((void*)out, (void*)in, len);
64  }
65  else
66  {
67  cpuvec<REAL_OUT> cp(fml::kernelfuns::CPLEN);
68  REAL_OUT *cp_d = cp.data_ptr();
69 
70  size_t top = (size_t) m*n;
71  for (size_t i=0; i<top; i+=fml::kernelfuns::CPLEN)
72  {
73  const size_t start = top - i;
74  const size_t copylen = std::min(fml::kernelfuns::CPLEN, start);
75  c->mem_cpu2gpu((void*)cp_d, (void*)(in + start), copylen);
76  arraytools::copy(copylen, cp_d, out + start);
77  }
78  }
79  }

93 
94  size_t top = (size_t) m*n;
95  for (size_t i=0; i<top; i+=fml::kernelfuns::CPLEN)
96  {
97  const size_t start = top - i;
98  const size_t copylen = std::min(fml::kernelfuns::CPLEN, start);

216  {
217  auto c = gpu_in.get_card();
