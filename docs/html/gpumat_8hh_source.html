252 template <typename REAL>
253 void fml::gpumat<REAL>::resize(len_t nrows, len_t ncols)
254 {
255  check_params(nrows, ncols);
256 
257  const size_t len = (size_t) nrows * ncols * sizeof(REAL);
258  const size_t oldlen = (size_t) this->m * this->n * sizeof(REAL);
259 
260  if (len == oldlen)
261  {
262  this->m = nrows;
263  this->n = ncols;
264  return;
265  }
266 
267  REAL *realloc_ptr;
268  realloc_ptr = (REAL*) this->c->mem_alloc(len);
269 
270  const size_t copylen = std::min(len, oldlen);
271  this->c->mem_gpu2gpu(realloc_ptr, this->data, copylen);
272  this->c->mem_free(this->data);
273  this->data = realloc_ptr;
274 
275  this->m = nrows;
276  this->n = ncols;
277 
278  dim_block = fml::kernel_launcher::dim_block2();
279  dim_grid = fml::kernel_launcher::dim_grid(this->m, this->n);
280 }

783 bool fml::gpumat<T>::operator==(const fml::gpumat<T> &x) const
784 {
785  if (this->m != x.nrows() || this->n != x.ncols())
786  return false;
787  else if (this->c->get_id() != x.get_card()->get_id())
788  return false;
789  else if (this->data == x.data_ptr())
790  return true;
791 
792  int all_eq = 1;
793  fml::gpuscalar<int> all_eq_gpu(c, all_eq);
794 
795  fml::kernelfuns::kernel_all_eq<<<dim_grid, dim_block>>>(this->m, this->n, this->data, x.data_ptr(), all_eq_gpu.data_ptr());
796 
797  all_eq_gpu.get_val(&all_eq);
798  this->c->check();
799 
800  return (bool) all_eq;
