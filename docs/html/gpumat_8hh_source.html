241 template <typename REAL>
242 void fml::gpumat<REAL>::resize(len_t nrows, len_t ncols)
243 {
244  check_params(nrows, ncols);
245 
246  const size_t len = (size_t) nrows * ncols * sizeof(REAL);
247  const size_t oldlen = (size_t) this->m * this->n * sizeof(REAL);
248 
249  if (len == oldlen)
250  {
251  this->m = nrows;
252  this->n = ncols;
253  return;
254  }
255 
256  REAL *realloc_ptr;
257  realloc_ptr = (REAL*) this->c->mem_alloc(len);
258 
259  const size_t copylen = std::min(len, oldlen);
260  this->c->mem_gpu2gpu(realloc_ptr, this->data, copylen);
261  this->c->mem_free(this->data);
262  this->data = realloc_ptr;
263 
264  this->m = nrows;
265  this->n = ncols;
266 
267  dim_block = fml::kernel_launcher::dim_block2();
268  dim_grid = fml::kernel_launcher::dim_grid(this->m, this->n);
269 }

771 template <typename T>
772 bool fml::gpumat<T>::operator==(const fml::gpumat<T> &x) const
773 {
774  if (this->m != x.nrows() || this->n != x.ncols())
775  return false;
776  else if (this->c->get_id() != x.get_card()->get_id())
777  return false;
778  else if (this->data == x.data_ptr())
779  return true;
780 
781  int all_eq = 1;
782  fml::gpuscalar<int> all_eq_gpu(c, all_eq);
783 
784  fml::kernelfuns::kernel_all_eq<<<dim_grid, dim_block>>>(this->m, this->n, this->data, x.data_ptr(), all_eq_gpu.data_ptr());
785 
786  all_eq_gpu.get_val(&all_eq);
787  this->c->check();
788 
789  return (bool) all_eq;
